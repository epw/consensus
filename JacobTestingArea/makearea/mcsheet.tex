\SECTION{The Principles}
\begin{multicols}{2}
\begin{itemize}
\item Riddle the world with hidden magic, unexpected connections, and
  unrecognized potential
\item Summon up symbols
\item Address yourself to the characters, not the players
\item Make your move, but misdirect, never speak its name
%\item Make your move, but never speak its name
\item Look down the sights
\item Name everyone, make everyone human
\item Ask provocative questions, build on the answers
\item Sometimes, give them exactly what they’re asking for
\item Be a fan of the players' characters
\item Think offscreen too
\item Sometimes, declaim decision-making
\item {\large\bf Don't be afraid to be obvious!}
\end{itemize}
\end{multicols}

\SECTION{Starting a Session}
\begin{itemize}
  \setlength\itemsep{-.4em}
\item Gather this page, PC summary sheet, Fronts, Flags, and
  Foreshadows
\item Look over and read out Flags, note which ones can be changed
\item Mark one Code off every Foreshadow Status Track
\item Distribute and resolve ``love letters''
\item Mark one Code off every Anchor Status Track
\item Note all slipping Status Tracks
\item Review any effects that last session were decided to be extended
  into this one
\item Perform moves triggered at start of session
\item Frame a scene and make a soft move
\end{itemize}

\SECTION{Scene-Framing Techniques}
\begin{itemize}
\item Pick 2-3 people, tell them where they are, make a soft move, and
  ask them why they're there
\end{itemize}
